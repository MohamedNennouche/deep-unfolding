  6def train_model(
  7    model: torch.nn.Module, 
  8    optimizer: torch.optim.Optimizer, 
  9    loss_func: torch.nn.Module, 
 10    total_itr: int, 
 11    solution: torch.Tensor, 
 12    num_batch: int
 13) -> Tuple[torch.nn.Module, List[float]]:
 14    """
 15    Train the given model using the specified optimizer and loss function.
 16
 17    Args:
 18        - model (torch.nn.Module): The model to be trained.
 19        - optimizer (torch.optim.Optimizer): The optimizer to use for training.
 20        - loss_func (torch.nn.Module): The loss function to use for training.
 21        - total_itr (int): The total number of iterations (generations) for training.
 22        - solution (torch.Tensor): The target solution tensor.
 23        - num_batch (int): The number of batches per iteration.
 24
 25    Returns:
 26        Tuple[torch.nn.Module, List[float]]: The trained model and the list of loss values per iteration.
 27
 28    """
 29    loss_gen = []
 30    for gen in range(total_itr):
 31        for i in range(num_batch):
 32            optimizer.zero_grad()
 33            x_hat, _ = model(gen + 1)
 34            loss = loss_func(x_hat, solution)
 35            loss.backward()
 36            optimizer.step()
 37            
 38            if i % 200 == 0:
 39                print("generation:", gen + 1, " batch:", i, "\t MSE loss:", loss.item())
 40        loss_gen.append(loss.item())
 41    return model, loss_gen

 44def evaluate_model(
 45    model: torch.nn.Module, 
 46    solution: torch.Tensor, 
 47    n: int, 
 48    bs: int, 
 49    total_itr: int, 
 50    device: torch.device = device
 51) -> List[float]:
 52    """
 53    Evaluate the model by calculating the mean squared error (MSE) between the solution and the model's predictions.
 54
 55    Args:
 56        - model (torch.nn.Module): The model to be evaluated.
 57        - solution (torch.Tensor): The target solution tensor.
 58        - n (int): The dimension of the solution.
 59        - bs (int): The batch size.
 60        - total_itr (int): The total number of iterations for evaluation.
 61        - device (torch.device, optional): The device to run the evaluation on. Defaults to device.
 62
 63    Returns:
 64        List[float]: A list of MSE values for each iteration.
 65    """
 66    norm_list = []
 67    for i in range(total_itr + 1):
 68        s_hat, _ = model(i)
 69        err = (torch.norm(solution.to(device) - s_hat.to(device)) ** 2).item() / (n * bs)
 70        norm_list.append(err)
 71    return norm_list

 74class SORNet(nn.Module):
 75    """Deep unfolded SOR with a constant step size.
 76
 77    Args:
 78        - init_val_SORNet (float): Initial value for inv_omega.
 79        - A (torch.Tensor): Matrix A of the linear system.
 80        - H (torch.Tensor): Matrix H.
 81        - bs (int): Batch size.
 82        - y (torch.Tensor): Solution of the linear equation.
 83        - device (torch.device): Device to run the model on ('cpu' or 'cuda').
 84
 85    Attributes:
 86        - device (torch.device): Device to run the model on.
 87        - inv_omega (nn.Parameter): Inverse of the relaxation parameter omega.
 88        - A (torch.Tensor): Matrix A of the linear system.
 89        - D (torch.Tensor): Diagonal matrix D.
 90        - L (torch.Tensor): Lower triangular matrix L.
 91        - U (torch.Tensor): Upper triangular matrix U.
 92        - H (torch.Tensor): Matrix H.
 93        - Dinv (torch.Tensor): Inverse of the diagonal matrix D.
 94        - bs (int): Batch size.
 95        - y (torch.Tensor): Solution of the linear equation.
 96    
 97    Methods:
 98        forward(num_itr: int) -> Tuple[torch.Tensor, List[torch.Tensor]]:
 99            Perform forward pass of the SORNet model.
100    """
101
102    def __init__(self, init_val_SORNet: float, A: torch.Tensor, H: torch.Tensor, bs: int, y: torch.Tensor, device: torch.device = device):
103        """
104        Initialize the SORNet model.
105
106        Args:
107            - init_val_SORNet (float): Initial value for inv_omega.
108            - A (torch.Tensor): Matrix A of the linear system.
109            - H (torch.Tensor): Matrix H.
110            - bs (int): Batch size.
111            - y (torch.Tensor): Solution of the linear equation.
112            - device (torch.device): Device to run the model on ('cpu' or 'cuda').
113
114        """
115        super(SORNet, self).__init__()
116        self.device = device
117        self.inv_omega = nn.Parameter(torch.tensor(init_val_SORNet, device=device))
118
119        A, D, L, U, _, _ = decompose_matrix(A)
120
121        self.A = A.to(device)
122        self.D = D.to(device)
123        self.L = L.to(device)
124        self.U = U.to(device)
125        self.H = H.to(device)
126        self.Dinv = torch.linalg.inv(D).to(device)
127        self.bs = bs
128        self.y = y.to(device)
129
130    def forward(self, num_itr: int) -> Tuple[torch.Tensor, List[torch.Tensor]]:
131        """
132        Perform forward pass of the SORNet model.
133
134        Args:
135            num_itr (int): Number of iterations.
136
137        Returns:
138            - torch.Tensor: Output tensor.
139            - list: List of intermediate results.
140
141        """
142        traj = []
143
144        invM = torch.linalg.inv(self.inv_omega * self.D + self.L)
145        s = torch.zeros(self.bs, self.H.size(1), device=self.device)
146        traj.append(s)
147        yMF = torch.matmul(self.y, self.H.T)
148        s = torch.matmul(yMF, self.Dinv)
149
150        for _ in range(num_itr):
151            temp = torch.matmul(s, (self.inv_omega - 1) * self.D - self.U) + yMF
152            s = torch.matmul(temp, invM)
153            traj.append(s)
154
155        return s, traj

171
172    Attributes:

174        - gamma (nn.Parameter): Gamma parameter for each iteration.
175        - omega (nn.Parameter): Omega parameter for each iteration.
176        - inv_omega (nn.Parameter): Inverse of the relaxation parameter omega.
177        - A (torch.Tensor): Matrix A of the linear system.
178        - D (torch.Tensor): Diagonal matrix D.
179        - L (torch.Tensor): Lower triangular matrix L.
180        - U (torch.Tensor): Upper triangular matrix U.
181        - H (torch.Tensor): Matrix H.
182        - Dinv (torch.Tensor): Inverse of the diagonal matrix D.
183        - bs (int): Batch size.
184        - y (torch.Tensor): Solution of the linear equation.
185
186    Methods:
187        forward(num_itr: int) -> Tuple[torch.Tensor, list[torch.Tensor]]:
188            Perform forward pass of the SOR_CHEBY_Net model.
189    """
190
191    def __init__(self, num_itr: int, init_val_SOR_CHEBY_Net_omega: float, init_val_SOR_CHEBY_Net_gamma: float, init_val_SOR_CHEBY_Net_alpha: float, A: torch.Tensor, H: torch.Tensor, bs: int, y: torch.Tensor, device: torch.device = device):
192        """
193        Initialize the SOR_CHEBY_Net model.
194
195        Args:
196            - num_itr (int): Number of iterations.
197            - init_val_SOR_CHEBY_Net_omega (float): Initial value for omega.
198            - init_val_SOR_CHEBY_Net_gamma (float): Initial value for gamma.
199            - init_val_SOR_CHEBY_Net_alpha (float): Initial value for inv_omega.
200            - A (torch.Tensor): Matrix A of the linear system.
201            - H (torch.Tensor): Matrix H.
202            - bs (int): Batch size.
203            - y (torch.Tensor): Solution of the linear equation.
204            - device (torch.device): Device to run the model on ('cpu' or 'cuda').
205
206        """
207        super(SOR_CHEBY_Net, self).__init__()
208        self.device = device
209        self.gamma = nn.Parameter(init_val_SOR_CHEBY_Net_gamma * torch.ones(num_itr, device=device))
210        self.omega = nn.Parameter(init_val_SOR_CHEBY_Net_omega * torch.ones(num_itr, device=device))
211        self.inv_omega = nn.Parameter(torch.tensor(init_val_SOR_CHEBY_Net_alpha, device=device))
212
213        A, D, L, U, _, _ = decompose_matrix(A)
214        self.A = A
215        self.D = D.to(device)
216        self.L = L.to(device)
217        self.U = U.to(device)
218        self.H = H.to(device)
219        self.Dinv = torch.linalg.inv(D).to(device)
220        self.bs = bs
221        self.y = y.to(device)
222
223    def forward(self, num_itr: int) -> Tuple[torch.Tensor, List[torch.Tensor]]:
224        """
225        Perform forward pass of the SOR_CHEBY_Net model.
226
227        Args:
228            num_itr (int): Number of iterations.
229
230        Returns:
231            - torch.Tensor: Output tensor.
232            - list: List of intermediate results.
233
234        """
235        traj = []
236
237        invM = torch.linalg.inv(self.inv_omega * self.D + self.L)
238        s = torch.zeros(self.bs, self.H.size(1), device=self.device)
239        s_new = torch.zeros(self.bs, self.H.size(1), device=self.device)
240        traj.append(s)
241        yMF = torch.matmul(self.y, self.H.T)
242        s = torch.matmul(yMF, self.Dinv)
243        s_present = s
244        s_old = torch.zeros_like(s_present)
245
246        for i in range(num_itr):
247            temp = torch.matmul(s, (self.inv_omega - 1) * self.D - self.U) + yMF
248            s = torch.matmul(temp, invM)
249
250            s_new = self.omega[i] * (self.gamma[i] * (s - s_present) + (s_present - s_old)) + s_old
251            s_old = s
252            s_present = s_new
253            traj.append(s_new)
254
255        return s_new, traj

275        - A (torch.Tensor): Matrix A of the linear system.
276        - D (torch.Tensor): Diagonal matrix D.
277        - L (torch.Tensor): Lower triangular matrix L.
278        - U (torch.Tensor): Upper triangular matrix U.
279        - H (torch.Tensor): Matrix H.

281        - bs (int): Batch size.
282        - y (torch.Tensor): Solution of the linear equation.
283    
284    Methods:
285        forward(num_itr: int) -> Tuple[torch.Tensor, list[torch.Tensor]]:
286            Perform forward pass of the AORNet model.
287    """
288
289    def __init__(self, init_val_AORNet_r: float, init_val_AORNet_omega: float, A: torch.Tensor, H: torch.Tensor, bs: int, y: torch.Tensor, device: torch.device = device):
290        """
291        Initialize the AORNet model.
292
293        Args:
294            - init_val_AORNet_r (float): Initial value for r.
295            - init_val_AORNet_omega (float): Initial value for omega.
296            - A (torch.Tensor): Matrix A of the linear system.
297            - H (torch.Tensor): Matrix H.
298            - bs (int): Batch size.
299            - y (torch.Tensor): Solution of the linear equation.
300            - device (torch.device): Device to run the model on ('cpu' or 'cuda').
301
302        """
303        super(AORNet, self).__init__()
304        self.device = device
305        self.r = nn.Parameter(torch.tensor(init_val_AORNet_r, device=device))
306        self.omega = nn.Parameter(torch.tensor(init_val_AORNet_omega, device=device))
307
308        A, D, L, U, _, _ = decompose_matrix(A)
309        self.A = A.to(device)
310        self.D = D.to(device)
311        self.L = L.to(device)
312        self.U = U.to(device)
313        self.H = H.to(device)
314        self.Dinv = torch.linalg.inv(D).to(device)
315        self.bs = bs
316        self.y = y.to(device)
317
318    def forward(self, num_itr: int) -> Tuple[torch.Tensor, list[torch.Tensor]]:
319        """
320        Perform forward pass of the AORNet model.
321
322        Args:
323            num_itr (int): Number of iterations.
324
325        Returns:
326            - torch.Tensor: Output tensor.
327            - list: List of intermediate results.
328
329        """
330        traj = []
331
332        invM = torch.linalg.inv(self.L - self.r * self.D)
333        N = (1 - self.omega) * self.D + (self.omega - self.r) * self.L + self.omega * self.U
334        s = torch.zeros(self.bs, self.H.size(1), device=self.device)
335        traj.append(s)
336        yMF = torch.matmul(self.y, self.H.T)
337        s = torch.matmul(yMF, self.Dinv)
338
339        for _ in range(num_itr):
340            s = torch.matmul(s, torch.matmul(invM, N)) + torch.matmul(yMF, invM)
341            traj.append(s)
342
343        return s, traj
